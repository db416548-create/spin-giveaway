<script>
let currentRotation = 0;

async function spin(){
  const res = await fetch("https://URL-KAMU.onrender.com/spin");
  const data = await res.json();

  if(data.result === "LIMIT"){
    document.getElementById("result").innerText = "Kamu sudah pernah spin";
    return;
  }

  if(data.result === "HABIS"){
    document.getElementById("result").innerText = "Hadiah sudah habis, coba lagi besok ya";
    return;
  }

  const map = {
    "HG": 36,
    "PRIME": 108,
    "DRIP": 180,
    "BONUS": 252,
    "ZONK": 324
  };

  const target = map[data.result] || 0;

  const spinDeg = 3600 + target;
  currentRotation += spinDeg;

  document.getElementById("wheel").style.transform = `rotate(${currentRotation}deg)`;

  setTimeout(()=>{
    document.getElementById("result").innerText =
      "Hasil: " + data.result + "
Kode: " + data.code;

    window.location.href =
      "https://wa.me/6282211243753?text=Kode%20saya%20" + data.code;
  },4000);
}
</script>
